    private static double getSumaWithWhile(int x)
    {
        double sum=1;
        double sumand=1;
        double precision =0.0001;
        int i=1;
        while(sumand>precision){
            double newSumAnd=sumand*x/i;
            sum+=newSumAnd;
            sumand=newSumAnd;
            i++;
        }
        return sum;
    }
    private static double getSumWithFor(int x)
    {
        int loopcount=500;
        double sum=1;
        double sumand=1;
        for(int i=1;i<loopcount;++i)
        {
            double newSumand=sumand*x/i;
            sum+=newSumand;
            sumand=newSumand;
        }
        return sum;
    }
}
--------------------Taylor Series For the Sin and Cos --------------

 public class Main
{
    public static void main(String[] args)
    {
        double x = Math.PI / 3; // 60 degrees

        // Sine
        double sinWithFor = getSinWithFor(x);
        double sinWithWhile = getSinWithWhile(x);

        System.out.println("Sin with for: " + sinWithFor);
        System.out.println("Sin with while: " + sinWithWhile);
        System.out.println("Real sin: " + Math.sin(x));

        // Cosine
        double cosWithFor = getCosWithFor(x);
        double cosWithWhile = getCosWithWhile(x);

        System.out.println("Cos with for: " + cosWithFor);
        System.out.println("Cos with while: " + cosWithWhile);
        System.out.println("Real cos: " + Math.cos(x));
    }

    // ---------- Sine ----------
    private static double getSinWithFor(double x)
    {
        int loopcount = 50;
        double sum = 0;
        double term = x; // first term x^(2*0+1)/(2*0+1)! = x
        int sign = 1;

        for (int i = 0; i < loopcount; i++)
        {
            sum += sign * term;
            term = term * x * x / ((2 * i + 2) * (2 * i + 3)); // next term
            sign *= -1;
        }
        return sum;
    }

    private static double getSinWithWhile(double x)
    {
        double sum = 0;
        double term = x;
        int sign = 1;
        double precision = 0.000001;
        int i = 0;

        while (Math.abs(term) > precision)
        {
            sum += sign * term;
            term = term * x * x / ((2 * i + 2) * (2 * i + 3));
            sign *= -1;
            i++;
        }
        return sum;
    }

    // ---------- Cosine ----------
    private static double getCosWithFor(double x)
    {
        int loopcount = 50;
        double sum = 0;
        double term = 1; // first term x^(0)/(0)! = 1
        int sign = 1;

        for (int i = 0; i < loopcount; i++)
        {
            sum += sign * term;
            term = term * x * x / ((2 * i + 1) * (2 * i + 2)); // next term
            sign *= -1;
        }
        return sum;
    }

    private static double getCosWithWhile(double x)
    {
        double sum = 0;
        double term = 1;
        int sign = 1;
        double precision = 0.000001;
        int i = 0;

        while (Math.abs(term) > precision)
        {
            sum += sign * term;
            term = term * x * x / ((2 * i + 1) * (2 * i + 2));
            sign *= -1;
            i++;
        }
        return sum;
    }
}
